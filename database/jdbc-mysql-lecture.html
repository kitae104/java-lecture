01 JDBC 아키텍처 — Java ↔ MySQL 연결 구조

03 소스 코드 — JDBC_Test.java
JDBC_Test.java JDBC 
package inhatc.cse.java.chap17.ex4_db;

import javax.swing.*;
import java.sql.*;

public class JDBC_Test {
    public static void main(String[] args) {

        try {
            // ① 드라이버 로드 ─────────────────────────────────────────
            // MySQL JDBC 드라이버 클래스를 JVM 메모리에 등록
            // ClassNotFoundException 발생 시 → JAR가 classpath에 없는 것
            Class.forName("com.mysql.cj.jdbc.Driver");

            // ② DB 연결 ───────────────────────────────────────────────
            // URL 형식: jdbc:mysql://호스트:포트/DB명
            // 3번째 파라미터: DB 비밀번호
            Connection conn = DriverManager.getConnection(
                    "jdbc:mysql://localhost:3306/javastudy",
                    "root", "1111");

            Statement stmt = conn.createStatement();

            // ③ UPDATE — B1 회원 정보 수정 ────────────────────────────
            String updateSql =
                    "UPDATE MEMBER " +
                    "SET    PW='888', " +
                    "       AGE=30, " +
                    "       NAME='최재석' " +
                    "WHERE  ID='B1'";
            System.out.println(updateSql);
            stmt.executeUpdate(updateSql);    // 반환값: 영향받은 행 수

            // ④ DELETE — 사용자 입력으로 ID 받아 삭제 ─────────────────
            // ⚠ SQL Injection 취약 구간 (문자열 직접 연결)
            String delId = JOptionPane.showInputDialog("지우고 싶은 ID");
            String deleteSql = "DELETE FROM MEMBER WHERE ID='" + delId + "'";
            System.out.println(deleteSql);
            stmt.executeUpdate(deleteSql);

            // ⑤ SELECT — 전체 조회 ────────────────────────────────────
            ResultSet rs = stmt.executeQuery("SELECT * FROM MEMBER");

            System.out.println("--------------------------------");
            while(rs.next()) {          // 다음 행이 있으면 true
                String id   = rs.getString("id");
                String pw   = rs.getString("pw");
                int    age  = rs.getInt("age");
                String name = rs.getString("name");
                System.out.println(id + "\t| " + pw + "\t| " + age + "\t| " + name);
            }
            System.out.println("--------------------------------");
            System.out.println("OK!");

            conn.close();    // ⑥ 연결 종료 — 반드시 호출

        } catch (ClassNotFoundException e) {
            System.out.println("JDBC 드라이버 로드 에러");
            e.printStackTrace();
        } catch (SQLException e) {
            System.err.println("DB 연결 오류 또는 쿼리 오류 입니다.");
            e.printStackTrace();
        }
    }
}
코드단계설명
Class.forName("com.mysql.cj.jdbc.Driver") ① 드라이버 MySQL 8.x 드라이버 클래스명. 구버전(com.mysql.jdbc.Driver)과 다르므로 주의. JAR 미포함 시 ClassNotFoundException 발생
DriverManager.getConnection(url, id, pw) ② 연결 URL 형식: jdbc:mysql://호스트:포트/DB명. 연결 실패 시 SQLException 발생. 3306은 MySQL 기본 포트
conn.createStatement() ② 연결 SQL을 실행할 Statement 객체 생성. 매개변수가 없는 정적 SQL에 사용. 동적 SQL은 PreparedStatement 권장
stmt.executeUpdate(sql) ③ DML INSERT · UPDATE · DELETE 실행. 반환값은 영향받은 행(row) 수(int). 0이면 조건에 맞는 행 없음
JOptionPane.showInputDialog() 입력 Swing 팝업 대화상자로 사용자 입력 수신. 콘솔 Scanner 없이 간단하게 입력값 획득 가능
stmt.executeQuery(sql) ④ SELECT SELECT 실행. ResultSet 반환 — 커서가 첫 행 이전에 위치. rs.next()로 이동하며 읽음
rs.next() ④ SELECT ResultSet 커서를 다음 행으로 이동. 행이 있으면 true, 없으면 false. while 루프로 전체 결과 순회
rs.getString("컬럼명") / rs.getInt() ④ SELECT 컬럼명(대소문자 구분 없음) 또는 인덱스(1부터)로 값 읽기. 타입에 따라 getString · getInt · getDouble 등 선택
conn.close() ⑤ 종료 DB 연결 반환. 미호출 시 연결 누수 발생. 실무에서는 try-with-resources 또는 finally 블록에서 처리
04 CRUD 쿼리 상세 분석 — INSERT · UPDATE · DELETE · SELECT

SELECT는 executeQuery()를 사용하고 ResultSet으로 결과를 받습니다. rs.next()로 커서를 이동하며 행을 순서대로 읽습니다.

// 전체 조회
ResultSet rs = stmt.executeQuery("SELECT * FROM MEMBER");

while(rs.next()) {
    String id   = rs.getString("id");   // 문자열 컬럼
    String pw   = rs.getString("pw");
    int    age  = rs.getInt("age");     // 정수 컬럼
    String name = rs.getString("name");
    System.out.println(id + "\t| " + pw + "\t| " + age + "\t| " + name);
}

📋 콘솔 출력 예시

ID     | PW    | AGE | NAME
--------------------------------
A1     | 111   | 25  | 홍길동
B1     | 888   | 30  | 최재석  ← UPDATE 반영됨
C1     | 333   | 22  | 이철수
--------------------------------
OK!

UPDATE는 기존 행의 데이터를 수정합니다. WHERE 절을 생략하면 전체 행이 수정되므로 반드시 조건을 지정해야 합니다.

String updateSql =
    "UPDATE MEMBER " +
    "SET    PW='888', " +       // 비밀번호 변경
    "       AGE=30,   " +       // 나이 변경
    "       NAME='최재석' " +    // 이름 변경
    "WHERE  ID='B1'";           // ← 조건 필수! 없으면 전체 수정

int affected = stmt.executeUpdate(updateSql);
// affected == 1 이면 1개 행 수정 성공
// affected == 0 이면 ID='B1' 인 행이 없음
⚠️ WHERE 절 누락 주의: WHERE가 없으면 MEMBER 테이블 모든 행의 PW, AGE, NAME이 '888', 30, '최재석'으로 변경됩니다. 실행 전 WHERE 조건을 반드시 확인하세요.

DELETE는 JOptionPane으로 입력받은 ID를 SQL 문자열에 직접 이어 붙입니다. 동작하지만 SQL Injection 취약점이 있어 실무에서는 절대 사용하지 않습니다.

// 사용자 입력으로 삭제할 ID 입력받기
String delId     = JOptionPane.showInputDialog("지우고 싶은 ID");
String deleteSql = "DELETE FROM MEMBER WHERE ID='" + delId + "'";

// delId = "C1" 이면 → DELETE FROM MEMBER WHERE ID='C1'
// delId = "' OR '1'='1" 이면 → 전체 삭제! ← SQL Injection

stmt.executeUpdate(deleteSql);
🚨 SQL Injection 취약점
입력값을 SQL 문자열에 직접 연결("..." + delId + "...")하면
악의적 입력 ' OR '1'='1 시 WHERE 조건이 항상 참이 되어 전체 행 삭제 가능.

해결책 → PreparedStatement 사용:
PreparedStatement ps = conn.prepareStatement("DELETE FROM MEMBER WHERE ID=?");
ps.setString(1, delId);
ps.executeUpdate();

코드에 주석으로 포함된 INSERT 예제입니다. 주석을 해제하면 새 회원을 추가할 수 있습니다.

// 현재 주석 처리된 INSERT 예제
// 아래 주석을 해제하면 회원 추가 실행

String insertSQL =
    "INSERT INTO MEMBER (ID, PW, AGE, NAME) "
  + "VALUES('E1', '333', 33, '박태환')";

stmt.executeUpdate(insertSQL);

// executeUpdate() 반환값 = 삽입된 행 수 (보통 1)
// PK 중복 시 SQLException 발생 → catch 블록으로 이동
💡 INSERT 순서: 이 코드에서는 UPDATE → DELETE → SELECT 순으로 실행됩니다. INSERT를 활성화하려면 UPDATE/DELETE 코드와 함께 실행 순서를 고려해야 합니다. 중복 PK로 인한 오류에 주의하세요.
05 ResultSet 커서 동작 원리

executeQuery()가 반환하는 ResultSet은 커서(cursor)를 사용합니다. 처음에는 첫 번째 행 이전에 위치하며, rs.next()를 호출할 때마다 한 행씩 이동합니다.

ResultSet 커서 이동
/* executeQuery() 직후 */
▶ 커서: [ Before First ]
rs.next() → true
Row 1: A1 | 111 | 25 | 홍길동
rs.next() → true
Row 2: B1 | 888 | 30 | 최재석
rs.next() → true
Row 3: C1 | 333 | 22 | 이철수
▶ rs.next() → false → while 루프 종료
메서드반환 타입설명
rs.next()boolean커서를 다음 행으로 이동. 행이 있으면 true, 없으면 false
rs.getString("컬럼명")String현재 행의 문자열 컬럼 값 반환. 컬럼명 또는 인덱스(1~) 사용 가능
rs.getInt("컬럼명")int현재 행의 정수 컬럼 값 반환
rs.getDouble("컬럼명")double실수 컬럼 값 반환 (FLOAT, DOUBLE, DECIMAL 타입)
rs.getDate("컬럼명")java.sql.Date날짜 컬럼 값 반환 (DATE, DATETIME 타입)
06 예외 처리 구조 분석

JDBC는 두 종류의 예외를 처리해야 합니다. ClassNotFoundException과 SQLException은 발생 원인이 다르므로 별도의 catch 블록으로 구분합니다.

🔴
ClassNotFoundException
Class.forName() 실행 시 발생.

원인: MySQL Connector JAR가 classpath에 없거나, 드라이버 클래스명 오타

해결: pom.xml 의존성 추가 또는 JAR 직접 추가 확인
🔴
SQLException
DB 연결 · 쿼리 실행 · ResultSet 읽기 시 발생.

원인: URL/ID/PW 오류, 네트워크 문제, SQL 문법 오류, PK 중복, 테이블 미존재

해결: e.getMessage()로 원인 확인 
try {
    Class.forName(...);          // ClassNotFoundException 가능
    Connection conn = ...;        // SQLException 가능
    stmt.executeUpdate(...);     // SQLException 가능
    stmt.executeQuery(...);      // SQLException 가능
    conn.close();                 // SQLException 가능

} catch (ClassNotFoundException e) {
    // 드라이버 JAR 미포함 → 프로그램 종료 전 로그 출력
    System.out.println("JDBC 드라이버 로드 에러");
    e.printStackTrace();

} catch (SQLException e) {
    // DB 오류 → e.getErrorCode(), e.getMessage()로 상세 원인 확인
    System.err.println("DB 연결 오류 또는 쿼리 오류 입니다.");
    e.printStackTrace();
}
⚠️ conn.close() 위치 문제: 현재 코드는 conn.close()가 try 블록 안에 있어, 쿼리 실행 중 예외가 발생하면 close()가 호출되지 않을 수 있습니다. 실무에서는 finally 블록이나 try-with-resources를 사용합니다.
✅ 개선: try-with-resources (자동 close) 
try (
    Connection conn = DriverManager.getConnection(url, id, pw);
    Statement  stmt = conn.createStatement();
    ResultSet  rs   = stmt.executeQuery("SELECT * FROM MEMBER")
) {
    while(rs.next()) { ... }
    // 블록 종료 시 rs, stmt, conn 순으로 자동 close() 호출
} catch (SQLException e) {
    e.printStackTrace();
}
07 전체 실행 흐름
1
드라이버 로드 — Class.forName()
JVM이 com.mysql.cj.jdbc.Driver 클래스를 메모리에 로드. 이 시점에 드라이버가 DriverManager에 자동 등록됨. MySQL 8.x는 자동 등록을 지원하므로 최신 버전에서는 생략 가능하지만 명시 권장
2
DB 연결 수립 — DriverManager.getConnection()
TCP 소켓으로 MySQL 서버(localhost:3306)에 연결. javastudy 데이터베이스 선택, root/1111로 인증. Connection 객체 반환 — 이후 모든 DB 작업의 시작점
3
Statement 생성 — conn.createStatement()
SQL을 서버에 전송할 Statement 객체 생성. 하나의 Statement로 여러 SQL을 순서대로 실행 가능
4
UPDATE 실행 — B1 회원 정보 수정
stmt.executeUpdate(updateSql)로 B1의 PW, AGE, NAME 수정. 콘솔에 실행된 SQL 출력 (System.out.println(updateSql))
5
팝업 입력 → DELETE 실행
JOptionPane.showInputDialog()로 삭제할 ID 입력 대기. 입력값을 SQL에 연결해 executeUpdate()로 해당 행 삭제
6
SELECT 전체 조회 → ResultSet 출력
executeQuery("SELECT * FROM MEMBER")로 현재 상태 조회. while(rs.next())로 모든 행을 순서대로 읽어 콘솔 출력. UPDATE/DELETE 결과가 반영된 최신 데이터 확인
7
연결 종료 — conn.close()
DB 연결 해제. "OK!" 출력 후 프로그램 종료. 실무에서는 ResultSet · Statement · Connection 순으로 각각 닫는 것이 안전

executeUpdate(): INSERT·UPDATE·DELETE → int(영향 행 수)

rs.next()로 한 행씩 이동.
행 없으면 false → while 루프 종료
🔒
PreparedStatement 권장
사용자 입력을 SQL에 직접 연결하면 SQL Injection 위험.
? 플레이스홀더 + setString() 사용
🧹
자원 반드시 해제
conn.close() 미호출 시 연결 누수.
