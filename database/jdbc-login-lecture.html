03소스 코드
Login.java GUI · 이벤트 
package inhatc.cse.java.chap17.ex5_login;

import javax.swing.*;
import java.awt.*;
import java.awt.event.ActionEvent;
import java.awt.event.ActionListener;
import java.sql.ResultSet;
import java.sql.SQLException;

public class Login extends JFrame implements ActionListener {

    // ── GUI 컴포넌트 필드 ────────────────────────────────────
    private JPanel         panBase, panCenter, panSouth;
    private JButton        btnLogin;
    private JLabel         lbl1, lbl2;
    private JTextField     tfID;
    private JPasswordField tfPW;   // 비밀번호 전용 — 입력시 ●●● 마스킹

    public Login(String title, int width, int height) {
        setTitle(title);
        setSize(width, height);
        setLocation(300, 300);
        setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
        setLayout(new BorderLayout());

        createPanBase();   // 루트 패널 생성 → 내부에서 Center/South 패널 생성
        setVisible(true);
    }

    // ── 루트 패널: 배경 빨강, BorderLayout ────────────────────
    private void createPanBase() {
        panBase = new JPanel();
        panBase.setLayout(new BorderLayout());
        panBase.setBackground(Color.RED);   // 레이아웃 디버깅용 배경색

        createPanCenter();
        panBase.add(panCenter, BorderLayout.CENTER);
        createPanSouth();
        panBase.add(panSouth, BorderLayout.SOUTH);

        add(panBase);   // JFrame에 루트 패널 추가
    }

    // ── 중앙 패널: GridLayout 2×2 — ID/PW 입력폼 ─────────────
    private void createPanCenter() {
        panCenter = new JPanel();
        panCenter.setLayout(new GridLayout(2, 2));   // 2행 2열
        panCenter.setBorder(BorderFactory.createEmptyBorder(10, 10, 10, 20));

        lbl1 = new JLabel("ID :", JLabel.CENTER);
        tfID = new JTextField("A1", 15);     // 초기값 "A1"

        lbl2 = new JLabel("PW :", JLabel.CENTER);
        tfPW = new JPasswordField("777", 15);  // 초기값 "777", 마스킹 표시

        // GridLayout은 add 순서대로 좌→우, 위→아래 채움
        panCenter.add(lbl1);  panCenter.add(tfID);
        panCenter.add(lbl2);  panCenter.add(tfPW);
    }

    // ── 하단 패널: 로그인 버튼 ────────────────────────────────
    private void createPanSouth() {
        panSouth = new JPanel();   // 기본 FlowLayout
        btnLogin = new JButton("로그인");
        btnLogin.addActionListener(this);
        panSouth.add(btnLogin);
    }

    public static void main(String[] args) {
        DB.init();   // ← DB 연결 먼저 초기화 (주석 해제 필요)
        new Login("Login", 300, 150);
    }

    // ── 버튼 이벤트 처리 ──────────────────────────────────────
    @Override
    public void actionPerformed(ActionEvent e) {
        Object obj = e.getSource();

        if (obj == btnLogin) {
            String id = tfID.getText();
            String pw = tfPW.getText();   // JPasswordField도 getText()로 읽기

            boolean check = checkIDPW(id, pw);

            if (check) {
                System.out.println("로그인 성공");
                this.dispose();   // 현재 창 닫기 (종료X) → 다음 폼 띄울 자리
            } else {
                System.out.println("로그인 실패");
                JOptionPane.showMessageDialog(null, "로그인 실패했습니다.");
                tfID.setText("");   tfPW.setText("");
                tfID.requestFocus();   // 포커스를 ID 입력창으로
            }
        }
    }

    // ── DB 인증 메서드 ────────────────────────────────────────
    private boolean checkIDPW(String id, String pw) {
        boolean check = false;
        // ⚠ SQL Injection 취약: 입력값을 직접 문자열 연결
        String sql = "SELECT * FROM MEMBER WHERE ID = '"
                   + id + "' AND PW='" + pw + "'";
        System.out.println(sql);
        ResultSet rs = DB.getResultSet(sql);   // DB 클래스에 위임

        try {
            if (rs.next()) {            // 해당 ID/PW 행이 있으면 → 로그인 성공
                System.out.println(rs.getString(1));
                check = true;
            } else {
                System.out.println("해당 사용자가 없습니다.");
                check = false;
            }
        } catch (SQLException ex) { ex.printStackTrace(); }

        return check;
    }
}

GridLayout(2, 2) GUI 2행 2열 격자. add() 순서대로 Row 1: lbl1·tfID, Row 2: lbl2·tfPW 배치. 모든 칸 크기 동일
createEmptyBorder(10,10,10,20) GUI panCenter 안쪽 여백(padding). 순서: 상·좌·하·우. 오른쪽 여백 20px로 입력창이 레이블에서 조금 떨어짐
DB.init() — main()에서 호출 DB GUI 창 생성 전에 DB 연결 초기화. 현재 주석 처리됨 — 실행 시 반드시 활성화해야 로그인 가능
this.dispose() 이벤트 현재 JFrame을 닫음. System.exit()과 달리 JVM은 종료되지 않음 → 로그인 성공 후 다음 화면으로 전환할 때 사용
DB.getResultSet(sql) DB SQL 실행을 DB 클래스에 위임. ResultSet 반환 — rs.next()로 해당 ID/PW 행 존재 여부 확인
rs.getString(1) DB 컬럼명 대신 인덱스(1=첫 번째 컬럼=ID) 사용. 로그인 성공 시 해당 ID 콘솔 출력
DB.java DB 유틸리티 
package inhatc.cse.java.chap17.ex5_login;

import java.sql.*;

public class DB {

    // ── static 필드: 클래스 전체에서 하나의 연결 공유 ─────────
    public static Connection conn;   // DB 연결 객체
    public static Statement  stmt;   // SQL 실행 객체

    // ── DB 초기화: 드라이버 로드 + 연결 수립 ─────────────────
    public static void init() {
        try {
            // Oracle 드라이버 (이 예제 코드의 원본)
            // MySQL로 전환 시 아래 두 줄 수정 필요
            Class.forName("oracle.jdbc.driver.OracleDriver");
            conn = DriverManager.getConnection(
                    "jdbc:oracle:thin:@127.0.0.1:1521:XE",
                    "kitae", "1111");

            // MySQL 사용 시 → 아래 코드로 교체
            // Class.forName("com.mysql.cj.jdbc.Driver");
            // conn = DriverManager.getConnection(
            //         "jdbc:mysql://localhost:3306/javastudy",
            //         "root", "1111");

            stmt = conn.createStatement();
            System.out.println("DB 연결 성공");
        } catch (ClassNotFoundException e) {
            System.out.println("JDBC 드라이버 로드 에러");
            e.printStackTrace();
        } catch (SQLException e) {
            System.err.println("DB 연결 오류 또는 쿼리 오류 입니다.");
            e.printStackTrace();
        }
    }

    // ── SELECT 전용: ResultSet 반환 ───────────────────────────
    public static ResultSet getResultSet(String sql) {
        try {
            Statement stmt = conn.createStatement();   // 매번 새 Statement 생성
            return stmt.executeQuery(sql);
        } catch (Exception e) {
            System.out.println(e);
            return null;
        }
    }

    // ── INSERT · UPDATE · DELETE 전용 ────────────────────────
    public static void executeQuery(String sql) {
        try {
            stmt.executeUpdate(sql);   // 클래스 필드 stmt 사용
        } catch (SQLException e) {
            e.printStackTrace();
        }
    }
}
코드분류설명
public static Connection conn 설계 static 필드로 선언 → DB.conn으로 어디서든 접근. 프로그램 전체에서 하나의 연결을 공유 (싱글턴 패턴 유사)
DB.init() 초기화 드라이버 로드 + Connection + Statement 초기화. main()에서 GUI 생성 전에 가장 먼저 호출해야 함
oracle.jdbc.driver.OracleDriver 주의 원본 코드는 Oracle DB 설정. MySQL 사용 시 드라이버 클래스명과 JDBC URL을 MySQL용으로 변경해야 함 (주석 참고)
getResultSet(sql) — 새 Statement 생성 SELECT 매 호출마다 conn.createStatement()로 새 Statement 생성. 클래스 필드 stmt와 별도 — ResultSet이 닫히지 않도록
executeQuery(sql) — 필드 stmt 사용 DML INSERT·UPDATE·DELETE 전용. 클래스 필드 stmt 재사용. 메서드명이 executeQuery이지만 실제는 executeUpdate() 호출 — 혼동 주의

🔍 인증에 사용되는 SQL
-- id="A1", pw="777" 입력 시 생성되는 SQL
SELECT * FROM MEMBER WHERE ID = 'A1' AND PW='777'

-- rs.next()가 true  → 해당 회원 존재 → 로그인 성공
-- rs.next()가 false → 해당 회원 없음 → 로그인 실패
🚨 SQL Injection 취약점 — checkIDPW() 내부
입력값을 문자열로 직접 연결하므로, 악의적 입력 시 인증 우회 가능합니다.

공격 예시: ID에 ' OR '1'='1 입력 →
생성 SQL: SELECT * FROM MEMBER WHERE ID='' OR '1'='1' AND PW='...'
→ WHERE 조건이 항상 참 → PW 모르고도 rs.next()가 true 반환 → 로그인 성공

해결책 → PreparedStatement 사용:
PreparedStatement ps = conn.prepareStatement("SELECT * FROM MEMBER WHERE ID=? AND PW=?");
ps.setString(1, id); ps.setString(2, pw);
ResultSet rs = ps.executeQuery();
05DB.java 설계 포인트 — static 유틸리티 패턴

DB.java는 모든 필드와 메서드를 static으로 선언합니다. new DB() 없이 DB.init(), DB.getResultSet()처럼 클래스 이름으로 직접 호출하는 유틸리티 클래스 패턴입니다.

🔗
static Connection 공유
DB.conn은 JVM 메모리에 하나만 존재. init()으로 한 번 연결하면 프로그램 종료까지 재사용 가능
🔀
메서드 역할 분리
getResultSet(): SELECT → ResultSet 반환
executeQuery(): INSERT·UPDATE·DELETE → 반환값 없음
⚠️
메서드명 혼동 주의
DB.executeQuery()는 이름과 달리 내부에서 stmt.executeUpdate() 호출 — DML 전용임을 기억할 것
📋
getResultSet의 Statement
매번 conn.createStatement()로 새 Statement 생성. 이전 ResultSet이 열려 있어도 충돌 없이 독립 실행 가능

06dispose() vs System.exit() — 창 닫기의 차이

로그인 성공 후 this.dispose()를 호출합니다. System.exit()과 달리 JVM을 종료하지 않으므로 다음 화면(메인 폼 등)을 연속으로 띄울 수 있습니다.

💀
System.exit(0)
JVM 전체 종료. 모든 스레드, 창, 리소스가 강제 해제.

로그인 성공 후 사용하면 프로그램 자체가 종료되므로 로그인 화면에서 사용 금지
✅
this.dispose()
현재 JFrame 창만 닫고 메모리 해제. JVM은 계속 실행.

dispose() 이후 new MainFrame() 등으로 다음 화면을 띄울 수 있음 
if (check) {
    System.out.println("로그인 성공");
    this.dispose();          // 로그인 창 닫기
    // new MainFrame();       ← 여기에 다음 화면 코드 추가
}
