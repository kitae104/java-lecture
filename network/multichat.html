04 소스 코드 (3개 파일)
ServerThread.java 핵심 클래스 
package inhatc.cse.java.chap17.ex3_multichat;

import java.io.*;
import java.net.Socket;
import java.util.ArrayList;

/**
 * 클라이언트 1개를 전담하는 스레드
 * - run()      : 해당 클라이언트의 메시지 수신 루프
 * - sendToAll(): 수신한 메시지를 threadList 전원에게 브로드캐스트
 */
public class ServerThread extends Thread {

    private Socket                    socket;
    private ArrayList<ServerThread> threadList;  // 모든 스레드 목록 (공유)
    private BufferedReader            in;
              BufferedWriter            out;   // sendToAllClients에서 접근 (패키지 접근)

    public ServerThread(Socket socket, ArrayList<ServerThread> threadList) {
        this.socket     = socket;
        this.threadList = threadList;
    }

    // ── 스레드 실행: 수신 루프 ──────────────────────────────
    @Override
    public void run() {
        try {
            in  = new BufferedReader(new InputStreamReader(socket.getInputStream()));
            out = new BufferedWriter(new OutputStreamWriter(socket.getOutputStream()));

            String inMsg;
            while(true) {
                inMsg = in.readLine();               // 담당 클라이언트 메시지 수신
                System.out.println("받은 메시지 : " + inMsg);

                if(inMsg.equalsIgnoreCase("bye")) {
                    threadList.remove(this);         // 목록에서 제거 후 종료
                    break;
                }
                sendToAllClients(inMsg);             // 전원 브로드캐스트
            }
        } catch (IOException e) {
            threadList.remove(this);               // 예외 발생 시에도 목록 제거
            e.printStackTrace();
        }
    }

    // ── 브로드캐스트: threadList 순회하며 전원 전송 ──────────
    private void sendToAllClients(String outMsg) {
        for (ServerThread st : threadList) {
            try {
                st.out.write(outMsg + "\n");   // 각 스레드의 out 필드에 직접 접근
                st.out.flush();
            } catch (IOException e) {
                e.printStackTrace();
            }
        }
    }
}
코드분류설명
extends Thread THREAD Thread 상속으로 run() 오버라이드. st.start() 호출 시 별도 스레드에서 실행
ArrayList<ServerThread> threadList THREAD ServerFrame에서 생성한 목록을 참조로 공유. 모든 스레드가 같은 리스트를 가리킴
BufferedWriter out (접근 제한자 없음) 설계 패키지 접근(default) — sendToAllClients에서 다른 스레드의 st.out에 접근하기 위해 의도적으로 private 제외
sendToAllClients(inMsg) THREAD threadList를 for-each 순회 → 모든 클라이언트에게 메시지 전달 = 브로드캐스트
threadList.remove(this) 설계 "bye" 수신 또는 IOException 시 자신을 목록에서 제거 → 이후 브로드캐스트 대상에서 제외
ServerFrame.java 
package inhatc.cse.java.chap17.ex3_multichat;

import javax.swing.*;
import java.awt.*;
import java.awt.event.ActionEvent;
import java.awt.event.ActionListener;
import java.io.*;
import java.net.ServerSocket;
import java.net.Socket;
import java.util.ArrayList;

public class ServerFrame extends JFrame implements ActionListener {

    private JPanel    panelCenter, panelSouth;
    private JTextField tf;
    private JButton    btn;
    private JTextArea  ta;

    private ServerSocket server = null;
    private BufferedWriter out  = null;

    // ── 핵심: 접속 클라이언트 스레드 목록 ────────────────
    private ArrayList<ServerThread> threadList = new ArrayList<>();

    public ServerFrame(String title, int width, int height, int x, int y) {
        setTitle(title);
        setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
        setLocation(x, y);
        setSize(width, height);
        setLayout(new BorderLayout());
        setCenter(); setSouth();
        setVisible(true);
        tf.requestFocus();
    }

    private void setCenter() {
        panelCenter = new JPanel(new BorderLayout());
        ta = new JTextArea(7, 20);
        ta.setEditable(false); ta.setLineWrap(true);
        panelCenter.add(new JScrollPane(ta,
            JScrollPane.VERTICAL_SCROLLBAR_ALWAYS,
            JScrollPane.HORIZONTAL_SCROLLBAR_NEVER));
        add(panelCenter, BorderLayout.CENTER);
    }

    private void setSouth() {
        panelSouth = new JPanel();
        tf = new JTextField(18); tf.addActionListener(this);
        btn = new JButton("전송"); btn.addActionListener(this);
        panelSouth.add(tf); panelSouth.add(btn);
        add(panelSouth, BorderLayout.SOUTH);
    }

    @Override
    public void actionPerformed(ActionEvent e) {
        if(e.getSource() == btn || e.getSource() == tf) {
            try {
                String outMsg = tf.getText();
                out.write(outMsg + "\n"); out.flush();
                ta.append("[서버] : " + outMsg + "\n");
                tf.setText(""); tf.requestFocus();
            } catch (IOException e1) { e1.printStackTrace(); }
        }
    }

    // ── 소켓 대기 루프: 클라이언트마다 ServerThread 생성 ──
    private void setSocket() {
        try {
            server = new ServerSocket(9999);
            ta.append("연결 대기중.....\n");

            while(true) {                               // 다중 클라이언트 대기
                Socket socket = server.accept();
                ta.append("연결 되었습니다.\n");

                ServerThread st = new ServerThread(socket, threadList);
                threadList.add(st);                      // 목록에 추가
                st.start();                               // 스레드 시작
            }
        } catch (IOException e) { e.printStackTrace(); }
    }

    public static void main(String[] args) {
        ServerFrame sf = new ServerFrame("Server", 300, 400, 400, 200);
        sf.setSocket();
    }
}
코드분류설명
ArrayList<ServerThread> threadList 핵심 접속한 모든 클라이언트 스레드를 보관. ServerThread 생성 시 이 목록을 참조로 전달
while(true) { server.accept() ... } 핵심 1:1 버전과의 차이점. 반복문으로 여러 클라이언트 수락 — 각각 ServerThread 생성
new ServerThread(socket, threadList) THREAD 접속한 소켓과 공유 threadList를 스레드에 전달
threadList.add(st); st.start() THREAD 목록 추가 후 start(). 추가 후 start 해야 sendToAll 시 포함됨에 주의

package inhatc.cse.java.chap17.ex3_multichat;

import javax.swing.*;
import java.awt.*;
import java.awt.event.ActionEvent;
import java.awt.event.ActionListener;
import java.io.*;
import java.net.Socket;

public class ChatFrame extends JFrame implements ActionListener {

    private JPanel    panelCenter, panelSouth;
    private JTextField tf;
    private JButton    btn;
    private JTextArea  ta;

    private Socket         socket = null;
    private BufferedReader  in     = null;

        setTitle(title);
        setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
        setLocation(x, y);
        setSize(width, height);
        setLayout(new BorderLayout());
        setCenter(); setSouth();
        setVisible(true);
        tf.requestFocus();
    }

    private void setCenter() {
        panelCenter = new JPanel(new BorderLayout());
        ta = new JTextArea(7, 20);
        ta.setEditable(false); ta.setLineWrap(true);
        panelCenter.add(new JScrollPane(ta,
            JScrollPane.VERTICAL_SCROLLBAR_ALWAYS,
            JScrollPane.HORIZONTAL_SCROLLBAR_NEVER));
        add(panelCenter, BorderLayout.CENTER);
    }

    private void setSouth() {
        panelSouth = new JPanel();
        tf = new JTextField(18); tf.addActionListener(this);
        btn = new JButton("전송"); btn.addActionListener(this);
        panelSouth.add(tf); panelSouth.add(btn);
        add(panelSouth, BorderLayout.SOUTH);
    }

    // ── 이벤트: 전송 버튼 or Enter ──────────────────────────
    @Override
    public void actionPerformed(ActionEvent e) {
        if(e.getSource() == btn || e.getSource() == tf) {
            try {
                String outMsg = tf.getText();
                out.write(outMsg + "\n"); out.flush();
                ta.append("[클라이언트] : " + outMsg + "\n");
                tf.setText(""); tf.requestFocus();
            } catch (IOException e1) { e1.printStackTrace(); }
        }
    }

    // ── 서버 연결 + 수신 루프 ────────────────────────────────
    private void runClient() {
        try {
            socket = new Socket("localhost", 9999);
            in  = new BufferedReader(new InputStreamReader(socket.getInputStream()));
            out = new BufferedWriter(new OutputStreamWriter(socket.getOutputStream()));

            String inMsg;
            while(true) {
                inMsg = in.readLine();                  // 서버로부터 브로드캐스트 수신
                ta.append("[서버] : " + inMsg + "\n");
            }
        } catch (IOException e) {
            e.printStackTrace();
        } finally {
            try { out.close(); in.close(); socket.close(); }
            catch (IOException e) { e.printStackTrace(); }
        }
    }

    public static void main(String[] args) {
        ChatFrame cf = new ChatFrame("Client", 300, 400, 700, 200);
        cf.runClient();
    }
}

05 실행 순서 (클라이언트 2명)
1
ServerFrame 실행
ServerFrame.main() → GUI 창 생성 → setSocket() 호출 → while(true) accept() 진입. 채팅창 "연결 대기중....." 표시
2
ChatFrame (Client A) 실행
첫 번째 클라이언트 실행 → ServerThread-A 생성 → threadList.add(A) → A.start() → 서버는 다시 accept() 대기
3
ChatFrame (Client B) 실행
두 번째 클라이언트 실행 → ServerThread-B 생성 → threadList.add(B) → B.start() → 이제 threadList = [A, B]
4
브로드캐스트 동작
A가 메시지 전송 → ServerThread-A.run()의 readLine() 수신 → sendToAllClients() 호출 → threadList[A, B] 전원에게 전달 → B의 화면에도 메시지 표시
5
클라이언트 종료 (bye)
"bye" 수신 → threadList.remove(this)로 목록 제거 → 스레드 종료. 나머지 클라이언트는 계속 채팅 가능
06 핵심 포인트 & 개선 방향
🧵
스레드 1개 = 클라이언트 1개
클라이언트가 N명이면 ServerThread도 N개. 각 스레드가 독립적으로 수신 루프 실행
📣
브로드캐스트 패턴
threadList 순회 → st.out.write(). 공유 목록을 통해 모든 클라이언트에게 메시지 전달
⚠️
동시성 주의
ArrayList는 thread-safe하지 않음. 실무에서는 CopyOnWriteArrayList 또는 synchronized 사용 권장
🔗
out 필드 패키지 접근
out을 private으로 선언하지 않아 sendToAllClients에서 st.out 직접 접근 가능
🧹
정상 종료 처리
bye 또는 IOException 모두에서 threadList.remove(this) 호출 → 끊긴 클라이언트 제거
🚀
확장 아이디어
닉네임 추가, 입장/퇴장 알림, 귓속말 기능, 채팅방 분리 등으로 발전 가능
